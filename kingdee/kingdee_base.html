# 5.1.2 兼容Oracle 使用p_AlterColumn 存储过程注意类型需要大写

EXEC p_AlterColumn 'CMK_RSM_POS','FSessionId','NVARCHAR(512)','null','1111','''''';

IF (ColumnType = 'DATETIME') THEN
v_ColumnType := 'DATE';
ELSIF (ColumnType = 'INT') THEN
v_ColumnType := 'NUMBER(10)';
ELSIF (ColumnType = 'SMALLINT') THEN
v_ColumnType := 'NUMBER(10)';    --ksql翻译oracle时，smallint翻译成了number10，这里必须保持一致
ELSIF (ColumnType = 'IMAGE') THEN
v_ColumnType := 'BLOB';
ELSIF (ColumnType = 'NTEXT') THEN
v_ColumnType := 'NCLOB';
ELSIF (ColumnType = 'BIGINT') THEN
v_ColumnType := 'NUMBER(19)';
ELSE
v_ColumnType := ColumnType;
END IF;
v_ColumnType := REPLACE(v_ColumnType, 'VARCHAR', 'VARCHAR2');
v_ColumnType := REPLACE(v_ColumnType, 'DECIMAL', 'NUMBER');
IF DefaultValue is not null AND DefaultValue <> 'NULL' THEN
IF DefaultValue = 'GETDATE()' THEN
v_defaultvalue := ' SYSDATE ';
ELSIF DefaultValue = 'NEWID()' THEN
v_defaultvalue := ' SYS_GUID() ';
ELSE
IF INSTR(DefaultValue, '{ts') = 1 THEN
v_defaultvalue := REPLACE(REPLACE(DefaultValue, '{ts', 'TO_DATE('), '}', ', ''YYYY-MM-DD HH24:MI:SS'')');
ELSE
v_defaultvalue := DefaultValue;
END IF;
END IF;
v_defaultsql := ' DEFAULT ' || v_defaultvalue;
END IF;
